        <h1>Calculadora de IMC</h1>

        <input type="number" id="peso" placeholder="Peso (kg)" step="0.1">

        <input type="number" id="altura" placeholder="Altura (m o cm)">

        <button onclick="calcularIMC()">Calcular</button>

        <div class="progress">

            <div class="progress-bar" id="barra"></div>

        </div>

        <div id="resultado"></div>

    </div>


    <script>

        function calcularIMC() {

            // Obtener valores

            const peso = parseFloat(document.getElementById("peso").value);

            let altura = parseFloat(document.getElementById("altura").value);


            // Validar entrada

            if (isNaN(peso) || isNaN(altura)) {

                alert("Ingresa valores válidos.");

                return;

            }


            // Convertir cm a metros si es necesario

            if (altura > 3) altura /= 100;


            // Calcular IMC

            const imc = peso / (altura * altura);

            const imcRedondeado = Math.round(imc * 100) / 100;


            // Clasificación

            let categoria;

            if (imc < 18.5) categoria = "Bajo peso ⚠️";

            else if (imc < 25) categoria = "Peso normal ✅";

            else if (imc < 30) categoria = "Sobrepeso ⚠️";

            else if (imc < 35) categoria = "Obesidad I ❌";

            else if (imc < 40) categoria = "Obesidad II ❌";

            else categoria = "Obesidad III ❌";


            // Mostrar resultado

            document.getElementById("resultado").innerHTML = 

                `IMC: ${imcRedondeado}<br>${categoria}`;

            document.getElementById("barra").style.width = `${Math.min(imc, 40)}%`;

        }

    </script>
